# Fungsi untuk mengklasifikasikan video dengan fitur AUTO-RETRY jika server sibuk
def ai_klasifikasi_video(judul_video):
    prompt = f"""
    Analisis judul video YouTube berikut dan tentukan 1 kata kategori/topik yang paling menggambarkan isi video tersebut.
    Contoh hasil yang diharapkan: Politik, Gadget, Game, Sepakbola, Edukasi, Hiburan, Kuliner, Bisnis.
    
    Judul Video: "{judul_video}"
    
    Jawab HANYA dengan nama kategorinya saja (maksimal 1 sampai 2 kata). Jangan berikan kalimat penjelasan, jangan pakai tanda kutip, dan jangan pakai tanda baca apa pun.
    """
    
    max_retries = 3  # Mencoba ulang maksimal 3 kali jika server 503/sibuk
    delay = 4.0      # Waktu tunggu awal (4 detik)
    
    for attempt in range(max_retries):
        try:
            response = client.models.generate_content(
                model='gemini-2.5-flash', 
                contents=prompt
            )
            hasil_kategori = response.text.strip() 
            if not hasil_kategori:
                return "Umum" 
            return hasil_kategori
            
        except Exception as e:
            # Cek apakah error disebabkan oleh server sibuk (503) atau kebanyakan request (429)
            if "503" in str(e) or "429" in str(e):
                print(f"   ⚠️ Server Google Sibuk (503). Menunggu {delay} detik sebelum mencoba lagi... (Percobaan {attempt + 1}/{max_retries})")
                time.sleep(delay)
                delay *= 2  # Jeda makin lama (4s -> 8s -> 16s) agar server stabil
            else:
                # Jika error jenis lain (misal masalah teks), langsung skip agar tidak looping terus
                print(f"Error AI saat mengklasifikasikan video: {e}")
                break
                
    return "Umum (ERROR AI)"
